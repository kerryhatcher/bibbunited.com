       ```typescript
       import {
         lexicalEditor,
         BlocksFeature,
         HeadingFeature,
         UploadFeature,
         HorizontalRuleFeature,
         BlockquoteFeature,
         LinkFeature,
         FixedToolbarFeature,
         EXPERIMENTAL_TableFeature,
       } from '@payloadcms/richtext-lexical'
       import { PullQuote } from '../blocks/PullQuote'
       import { Callout } from '../blocks/Callout'
       import { Embed } from '../blocks/Embed'

       export const richTextEditor = lexicalEditor({
         features: ({ defaultFeatures }) => [
           ...defaultFeatures,
           // D-10: headings (h2, h3, h4 -- h1 reserved for page title)
           HeadingFeature({ enabledHeadingSizes: ['h2', 'h3', 'h4'] }),
           // D-06: inline images with captions and alt text
           UploadFeature({
             collections: {
               media: {
                 fields: [
                   {
                     name: 'caption',
                     type: 'text',
                     label: 'Caption',
                   },
                 ],
               },
             },
           }),
           // D-05, D-07: pull quotes, callouts, embeds as blocks
           BlocksFeature({
             blocks: [PullQuote, Callout, Embed],
           }),
           // D-09: horizontal rule / section dividers
           HorizontalRuleFeature(),
           // D-08: table support for structured data
           EXPERIMENTAL_TableFeature(),
           // Fixed toolbar for better editor UX
           FixedToolbarFeature(),
         ],
       })
       ```
       Note: `defaultFeatures` already includes bold, italic, ordered/unordered lists, links, and blockquote (D-10 standard formatting). The spread `...defaultFeatures` preserves these. HeadingFeature is explicitly added to restrict heading levels. Check the actual Payload 3.x API -- if `defaultFeatures` is not a function parameter, adapt accordingly.       
       ```typescript
       import type { CollectionConfig } from 'payload'
       import { slugField } from '../fields/slug'
       import { ctaFields } from '../fields/cta'
       import { richTextEditor } from '../editors/richText'

       export const Pages: CollectionConfig = {
         slug: 'pages',
         admin: {
           useAsTitle: 'title',
           defaultColumns: ['title', 'slug', '_status', 'updatedAt'],
         },
         versions: {
           drafts: {
             autosave: {
               interval: 1500,
             },
             schedulePublish: true,
           },
         },
         access: {
           read: ({ req }) => {
             if (req.user) return true
             return { _status: { equals: 'published' } }
           },
         },
         fields: [
           {
             name: 'title',
             type: 'text',
             required: true,
           },
           slugField,
           {
             name: 'content',
             type: 'richText',
             required: true,
             editor: richTextEditor,
           },
           ctaFields,
         ],
       }
       ```
       Note: D-04 specifies flat hierarchy (no parent/child) -- there is no `parent` relationship field. This is intentional.
