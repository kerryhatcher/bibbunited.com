**1. Create `src/components/ui/Button.tsx`** (per D-14):

```tsx
import React from 'react'

type ButtonVariant = 'primary' | 'secondary'

interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant
  href?: string
  children: React.ReactNode
}

export function Button({
  variant = 'primary',
  children,
  className = '',
  href,
  ...props
}: ButtonProps) {
  const base =
    'inline-flex items-center justify-center font-heading font-bold uppercase tracking-wide px-6 min-h-[44px] transition-colors focus:outline-none focus:ring-2 focus:ring-accent focus:ring-offset-2'

  const variants: Record<ButtonVariant, string> = {
    primary: 'bg-accent text-text-on-accent hover:brightness-90',
    secondary:
      'border-2 border-accent text-accent bg-transparent hover:bg-accent hover:text-text-on-accent',
  }

  const classes = `${base} ${variants[variant]} ${className}`

  if (href) {
    return (
      <a href={href} className={classes}>
        {children}
      </a>
    )
  }

  return (
    <button className={classes} {...props}>
      {children}
    </button>
  )
}
```

Key details:
- 0px border-radius (Tailwind default with no rounded- class) per D-13
- min-h-[44px] for WCAG touch targets per DSGN-05
- focus:ring-2 focus:ring-accent focus:ring-offset-2 for keyboard nav per DSGN-05
- font-heading + font-bold + uppercase for protest-poster aesthetic per D-14
- Primary: solid bg-accent background; Secondary: border-2 border-accent outline with hover fill

**2. Create `src/components/ui/Card.tsx`** (per D-15):

```tsx
import React from 'react'

interface CardProps {
  imageSrc?: string
  imageAlt?: string
  children: React.ReactNode
  className?: string
  href?: string
}

export function Card({
  imageSrc,
  imageAlt = '',
  children,
  className = '',
  href,
}: CardProps) {
  const cardClasses = `border border-border bg-bg-dominant overflow-hidden transition-colors hover:border-t-4 hover:border-t-accent ${className}`

  const content = (
    <>
      {imageSrc && (
        <div className="w-full aspect-video overflow-hidden">
          <img
            src={imageSrc}
            alt={imageAlt}
            className="w-full h-full object-cover"
          />
        </div>
      )}
      <div className="p-6">{children}</div>
    </>
  )

  if (href) {
    return (
      <a
        href={href}
        className={`block ${cardClasses} focus:outline-none focus:ring-2 focus:ring-accent focus:ring-offset-2`}
      >
        {content}
      </a>
    )
  }

  return <article className={cardClasses}>{content}</article>
}
```

Key details:
- 0px border-radius (no rounded- class) per D-13
- Full-bleed image top with aspect-video per D-15
- 1px border using border-border token
- Hover: top border changes to 4px accent per UI-SPEC hover states
- Semantic <article> element per DSGN-05
- Focus ring on linked cards for keyboard nav

**3. Create `src/components/ui/Section.tsx`** (per D-16):

```tsx
import React from 'react'

interface SectionProps {
  children: React.ReactNode
  className?: string
  variant?: 'default' | 'dark'
}

export function Section({
  children,
  className = '',
  variant = 'default',
}: SectionProps) {
  const bg = variant === 'dark' ? 'bg-bg-secondary' : ''
  const text = variant === 'dark' ? 'text-text-on-dark' : ''

  return (
    <section
      className={`py-8 sm:py-12 px-4 sm:px-6 lg:px-8 ${bg} ${text} ${className}`}
    >
      <div className="mx-auto max-w-7xl">{children}</div>
    </section>
  )
}
```

Key details:
- Generous vertical padding: py-8 (32px) mobile, sm:py-12 (48px) tablet+ per D-16
- max-w-7xl (1280px) content container
- Dark variant for navy background sections
- Responsive horizontal padding
