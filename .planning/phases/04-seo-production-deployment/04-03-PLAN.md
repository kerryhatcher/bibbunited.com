Per D-15: GitHub Actions CI/CD that builds on push to main, tags with commit SHA, pushes to GHCR, updates argocd/ manifests.

Create `.github/workflows/build-deploy.yml` (NEW):
```yaml
name: Build and Deploy

on:
  push:
    branches: [main]
    paths-ignore:
      - '*.md'
      - '.planning/**'
      - '.claude/**'

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    permissions:
      contents: write
      packages: write
    steps:
      - uses: actions/checkout@v4

      - name: Log in to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build and push Docker image
        uses: docker/build-push-action@v6
        with:
          context: .
          push: true
          tags: |
            ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:sha-${{ github.sha }}
            ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest
          build-args: |
            DATABASE_URI=${{ secrets.BUILD_DATABASE_URI }}
            PAYLOAD_SECRET=${{ secrets.BUILD_PAYLOAD_SECRET }}

      - name: Update image tag in manifests
        run: |
          NEW_TAG="sha-${{ github.sha }}"
          sed -i "s|image: .*bibbunited.*|image: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${NEW_TAG}|" argocd/dev/deployment.yaml
          sed -i "s|image: .*bibbunited.*|image: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${NEW_TAG}|" argocd/prod/deployment.yaml

      - name: Commit manifest changes
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add argocd/
          git diff --staged --quiet || git commit -m "deploy: update image to sha-${{ github.sha }} [skip ci]"
          git push
```

Key details:
- `paths-ignore` prevents CI from re-triggering on markdown or planning file changes
- `[skip ci]` in the commit message prevents infinite loop (per Pitfall 7 from research)
- `permissions: contents: write` allows pushing manifest changes back
- `permissions: packages: write` allows pushing to GHCR
- Build args pass DATABASE_URI and PAYLOAD_SECRET for Payload build-time validation
