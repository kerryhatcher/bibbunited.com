Create the shared helpers file with these exact exports:

```typescript
import * as fs from 'fs';
import * as path from 'path';

export interface AuditScore {
  route: string;
  viewport: string;
  performance: number;
  accessibility: number;
  bestPractices: number;
  seo: number;
  axeViolations: number;
  axeIncomplete: number;
  pass: boolean;
}

export interface AxeViolationSummary {
  id: string;
  impact: string;
  description: string;
  nodes: number;
}

export const AUDIT_ROUTES = [
  '/',
  '/news',
  '/news/bibb-county-school-budget-2025',  // seeded news article slug
  '/contact-officials',
  '/meetings',
  '/about',  // CMS page /[slug]
];

export const RESULTS_DIR = path.join(process.cwd(), 'e2e', 'audit', 'results');

export const LIGHTHOUSE_THRESHOLDS = {
  performance: 95,
  accessibility: 95,
  'best-practices': 95,
  seo: 95,
};

export function ensureResultsDir(): void {
  fs.mkdirSync(RESULTS_DIR, { recursive: true });
}

export function writeJsonResult(filename: string, data: unknown): void {
  ensureResultsDir();
  fs.writeFileSync(
    path.join(RESULTS_DIR, filename),
    JSON.stringify(data, null, 2),
  );
}

export function summarizeAxeResults(results: { violations: any[]; passes: any[]; incomplete: any[] }) {
  return {
    violations: results.violations.length,
    passes: results.passes.length,
    incomplete: results.incomplete.length,
    details: results.violations.map((v: any) => ({
      id: v.id,
      impact: v.impact,
      description: v.description,
      nodes: v.nodes.length,
    })),
  };
}

export function generateAuditMarkdown(scores: AuditScore[]): string {
  const date = new Date().toISOString().split('T')[0];
  let md = '# Audit Results\n\n';
  md += `**Date:** ${date}\n`;
  md += `**Threshold:** 95+ Lighthouse (all categories), 0 axe-core violations\n`;
  md += `**Routes audited:** ${AUDIT_ROUTES.length}\n`;
  md += `**Viewports:** 5 (320x568, 375x667, 768x1024, 1024x768, 1440x900)\n\n`;

  md += '## Summary\n\n';
  const passing = scores.filter(s => s.pass).length;
  const total = scores.length;
  md += `**Overall:** ${passing}/${total} audits passing\n\n`;

  md += '## Scores by Route\n\n';
  md += '| Route | Viewport | Perf | A11y | BP | SEO | Axe Violations | Status |\n';
  md += '|-------|----------|------|------|----|-----|----------------|--------|\n';
  for (const s of scores) {
    const status = s.pass ? 'PASS' : 'FAIL';
    md += `| ${s.route} | ${s.viewport} | ${s.performance} | ${s.accessibility} | ${s.bestPractices} | ${s.seo} | ${s.axeViolations} | ${status} |\n`;
  }

  md += '\n## Future Polish\n\n';
  md += '_Items identified during audit that could be improved beyond the pass/fail threshold:_\n\n';
  md += '- (To be populated after audit execution)\n';

  return md;
}
```

Note: The `AUDIT_ROUTES` array includes a seeded news article slug. The executor MUST verify the actual seeded slug by checking `src/seed.ts` and adjust accordingly. If the slug differs, update it.

**File 1: `e2e/audit/lighthouse.spec.ts`**

Create Lighthouse audit spec per D-01, D-05, D-08, D-10, D-12. Key requirements:
- Uses `playwright-lighthouse` `playAudit()` with custom chromium fixture that has `--remote-debugging-port`
- Worker-scoped fixture: port = 9222 + parallelIndex to avoid port conflicts (Pitfall 2)
- Tests each route from AUDIT_ROUTES at the current viewport project
- Thresholds: `{ performance: 95, accessibility: 95, 'best-practices': 95, seo: 95 }` (D-05)
- Disables Lighthouse throttling for consistent local results (Anti-pattern from research)
- Writes JSON results per route per viewport to `e2e/audit/results/lighthouse-{route}-{viewport}.json` using `writeJsonResult` (D-12)
- Uses desktop config with `screenEmulation: { disabled: true }` so Playwright viewport applies (Pitfall 5)

Structure:
```typescript
import { chromium } from 'playwright';
import { test as base } from '@playwright/test';
import { playAudit } from 'playwright-lighthouse';
import { AUDIT_ROUTES, LIGHTHOUSE_THRESHOLDS, writeJsonResult } from '../helpers/audit-helpers';

const test = base.extend<{}, { port: number; auditBrowser: import('playwright').Browser }>({
  port: [async ({}, use, workerInfo) => {
    const port = 9222 + workerInfo.workerIndex;
    await use(port);
  }, { scope: 'worker' }],
  auditBrowser: [async ({ port }, use) => {
    const browser = await chromium.launch({
      args: [`--remote-debugging-port=${port}`],
    });
    await use(browser);
    await browser.close();
  }, { scope: 'worker' }],
});

const lighthouseConfig = {
  extends: 'lighthouse:default',
  settings: {
    formFactor: 'desktop' as const,
    screenEmulation: { disabled: true },
    throttling: {
      rttMs: 0,
      throughputKbps: 0,
      requestLatencyMs: 0,
      downloadThroughputKbps: 0,
      uploadThroughputKbps: 0,
      cpuSlowdownMultiplier: 1,
    },
    onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
  },
};

for (const route of AUDIT_ROUTES) {
  const routeName = route === '/' ? 'homepage' : route.replace(/\//g, '-').slice(1);
  test(`Lighthouse audit: ${route}`, async ({ auditBrowser, port }, testInfo) => {
    const page = await auditBrowser.newPage();
    await page.goto(`http://localhost:3000${route}`);
    await page.waitForLoadState('networkidle');

    await playAudit({
      page,
      port,
      thresholds: LIGHTHOUSE_THRESHOLDS,
      config: lighthouseConfig,
    });

    // Note: playAudit throws if thresholds not met.
    // For JSON output, we need to capture scores separately.
    // Use lighthouse directly for score capture:

    await page.close();
  });
}
```

IMPORTANT: The `playAudit` function from playwright-lighthouse throws on threshold failure but does not return scores in an easy-to-capture format. The executor should check the actual API -- if `playAudit` returns audit results, capture them. If not, the executor has discretion to either:
1. Use `playAudit` for threshold enforcement and accept pass/fail as the output
2. Import `lighthouse` directly alongside `playAudit` for score capture into JSON

The executor should test the actual behavior and adapt accordingly.

**File 2: `e2e/audit/accessibility.spec.ts`**

Create axe-core accessibility spec per D-02, D-06, D-08, D-10, D-12. Key requirements:
- Uses standard Playwright `test` and `page` fixture (no custom browser needed)
- Tests each route from AUDIT_ROUTES at the current viewport project
- Uses `AxeBuilder({ page }).withTags(['wcag2a', 'wcag2aa', 'wcag21aa']).analyze()`
- Asserts `results.violations` is empty array -- zero violations at any severity (D-06)
- Waits for `networkidle` before scanning (Pitfall 3)
- Writes JSON results to `e2e/audit/results/axe-{route}-{viewport}.json` using `writeJsonResult` (D-12)
- Writes violation summary using `summarizeAxeResults` for readable output

Structure:
```typescript
import { test, expect } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';
import { AUDIT_ROUTES, writeJsonResult, summarizeAxeResults } from '../helpers/audit-helpers';

for (const route of AUDIT_ROUTES) {
  const routeName = route === '/' ? 'homepage' : route.replace(/\//g, '-').slice(1);
  test(`Axe accessibility: ${route}`, async ({ page }, testInfo) => {
    await page.goto(route);
    await page.waitForLoadState('networkidle');

    const results = await new AxeBuilder({ page })
      .withTags(['wcag2a', 'wcag2aa', 'wcag21aa'])
      .analyze();

    const filename = `axe-${routeName}-${testInfo.project.name}.json`;
    writeJsonResult(filename, summarizeAxeResults(results));

    expect(
      results.violations,
      `Axe violations on ${route} at ${testInfo.project.name}: ${JSON.stringify(results.violations.map(v => ({ id: v.id, impact: v.impact, nodes: v.nodes.length })))}`
    ).toEqual([]);
  });
}
```
