    1. Create `src/hooks/formatSlug.ts` with the slug formatting utility and Payload field hook:
       ```typescript
       import type { FieldHook } from 'payload'

       export const formatSlug = (val: string): string =>
         val
           .replace(/ /g, '-')
           .replace(/[^\w-]+/g, '')
           .toLowerCase()

       export const formatSlugHook: FieldHook = ({ data, operation, value }) => {
         if (operation === 'create' || !value) {
           const fallback = data?.title
           if (fallback && typeof fallback === 'string') {
             return formatSlug(fallback)
           }
         }
         return value
       }
       ```

    2. Create `src/fields/slug.ts` with the reusable slug field (per D-02: auto-generated from title, manual override):
